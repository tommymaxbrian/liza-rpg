function makeTasks(){
  var d=new Date(),seed=d.getDate()+d.getMonth()*31+d.getFullYear(),res=[];
  CATS.forEach(function(c,i){
    var pool=POOL[c.id],idx=(seed+i*7)%pool.length;
    res.push(Object.assign({},pool[idx],{category:c.id}));
  });
  var ex=[
    Object.assign({},POOL.body[(seed+2)%POOL.body.length],{category:"body"}),
    Object.assign({},POOL.creative[(seed+3)%POOL.creative.length],{category:"creative"})
  ].filter(function(t){return !res.filter(function(r){return r.id===t.id;}).length;});
  return res.concat(ex.slice(0,2));
}

function toggleTask(id){

  S.done=was?S.done.filter(function(x){return x!==id;}).slice():S.done.concat([id]);
  sl("liza_done_"+tkey(),S.done);
  var prev=S.xp;
  S.xp=Math.max(0,S.xp+(was?-10:10));
  sl("liza_xp",S.xp);
  if(!was){
    UNLOCKS.forEach(function(u){
      if(prev<u.xp&&S.xp>=u.xp&&S.uins.indexOf(u.ii)<0){
        S.uins=S.uins.concat([u.ii]);
        sl("liza_ins",S.uins);
        S.popup=INSIGHTS[u.ii];
      }
    });
  }
  render();
}
