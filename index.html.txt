function todayKey(){return new Date().toISOString().split("T")[0];}
function weekKey(){const d=new Date(),day=d.getDay(),diff=d.getDate()-day+(day===0?-6:1);return new Date(new Date().setDate(diff)).toISOString().split("T")[0];}
function ls(k,d){try{const v=localStorage.getItem(k);return v!==null?JSON.parse(v):d;}catch{return d;}}
function ss(k,v){try{localStorage.setItem(k,JSON.stringify(v));}catch{}}

function getDayTasks(){
  const today=new Date(),seed=today.getDate()+today.getMonth()*31+today.getFullYear();
  const result=[];
  CATS.forEach((cat,ci)=>{
    const pool=ALL_TASKS[cat.id];
    const idx=(seed+ci*7)%pool.length;
    result.push({...pool[idx],category:cat.id});
  });
  const extras=[
    {...ALL_TASKS.body[(seed+2)%ALL_TASKS.body.length],category:"body"},
    {...ALL_TASKS.creative[(seed+3)%ALL_TASKS.creative.length],category:"creative"},
    {...ALL_TASKS.discipline[(seed+1)%ALL_TASKS.discipline.length],category:"discipline"},
  ].filter(t=>!result.find(r=>r.id===t.id));
  return [...result,...extras.slice(0,2)];
}

function catInfo(id){return CATS.find(c=>c.id===id);}
const typeLabel={micro:"быстро",daily:"сегодня"};
const typeClr={micro:["#9FE1CB","#085041"],daily:["#B5D4F4","#0C447C"]};

function toggleTask(id){
  const isDone=state.done.includes(id);
  state.done=isDone?state.done.filter(d=>d!==id):[...state.done,id];
  ss("liza_done_"+todayKey(),state.done);
  const delta=isDone?-10:10;
  const prevXp=state.xp;
  state.xp=Math.max(0,state.xp+delta);
  ss("liza_xp",state.xp);
  if(!isDone){
    UNLOCKS.forEach(u=>{
      if(prevXp<u.xp&&state.xp>=u.xp&&!state.unlockedInsights.includes(u.iIdx)){
        state.unlockedInsights=[...state.unlockedInsights,u.iIdx];
        ss("liza_insights",state.unlockedInsights);
        state.newInsight=INSIGHTS[u.iIdx];
      }
    });
  }
  render();
}

function toggleWeekly(id){
  state.weekDone=state.weekDone.includes(id)?state.weekDone.filter(d=>d!==id):[...state.weekDone,id];
  ss("liza_week_"+weekKey(),state.weekDone);
  render();
}

function setTab(t){state.tab=t;render();}
function closeInsight(){state.newInsight=null;render();}

function saveDiary(v){
  state.diary=v;
  ss("liza_diary_"+todayKey(),v);
}

    const wPct=WEEKLY.length?Math.round(weekDone.length/WEEKLY.length*100):0;
    html+=`<div class="weekly-progress-wrap">
      <div class="xp-label"><span>${weekDone.length}/${WEEKLY.length} выполнено</span><span>${wPct}%</span></div>
      <div class="xp-bar"><div class="xp-fill" style="width:${wPct}%"></div></div>
    </div>`;
    html+=WEEKLY.map(t=>{
      const cat=catInfo(t.cat);
      const isDone=weekDone.includes(t.id);
      return `<div class="task-item ${isDone?"done":""}" onclick="toggleWeekly('${t.id}')">
        <div class="check ${isDone?"checked":""}"></div>
        <div style="flex:1">
          <div class="badges">
            <span class="badge" style="background:${cat?.color};color:${cat?.text}">${cat?.icon} ${cat?.label}</span>
            <span class="badge" style="background:#f0eeea;color:#888">до ${t.deadline}</span>
          </div>
          <div class="task-text">${t.text}</div>
          ${!isDone?'<div class="xp-hint">+30 XP</div>':''}
        </div>
      </div>`;
    }).join("");
  }

  if(tab==="char"){
    html+=`<div class="card">
      <div style="display:flex;align-items:center;gap:14px;margin-bottom:14px">
        <div style="width:52px;height:52px;border-radius:50%;background:#eeedfe;display:flex;align-items:center;justify-content:center;font-size:22px">✦</div>
        <div><div style="font-size:16px;font-weight:600">Лиза</div><div style="font-size:13px;color:#534ab7">${charTrait.label} · Уровень ${level}</div></div>
      </div>
      <div style="font-size:13px;color:#666;line-height:1.6;margin-bottom:12px">${charTrait.desc}</div>
      <div class="xp-bar"><div class="xp-fill" style="width:${xpInLevel}%"></div></div>
      <div style="font-size:11px;color:#bbb;text-align:right;margin-top:4px">${xpInLevel}/100 XP</div>
    </div>
    <div class="card"><div class="card-title">Цели персонажа</div>
      ${["Уверенный человек открытый к новому","Идёт к цели мудро и рассудительно","Находит коннект с людьми видит свой путь","Переключается между разными режимами","Автономная и опирается на себя"].map(g=>`
        <div class="goal-row"><span class="goal-star">✦</span><span class="goal-text">${g}</span></div>`).join("")}
    </div>
    <div class="card"><div class="card-title">Разблокировки</div>
      ${UNLOCKS.map(u=>{
        const d=xp>=u.xp;
        return `<div class="unlock-row">
          <div class="unlock-circle" style="background:${d?"#5c9e2e":"#f0eeea"};color:${d?"#fff":"#bbb"}">${d?"✓":"🔒"}</div>
          <div><div style="font-size:13px;color:${d?"#1a1a1a":"#bbb"}">${u.label}</div><div style="font-size:11px;color:#bbb">${u.xp} XP</div></div>
        </div>`;}).join("")}
    </div>
    <div class="card"><div class="card-title">Прокачка по категориям</div>
      ${CATS.map(cat=>{
        const cd=tasks.filter(t=>t.category===cat.id&&done.includes(t.id)).length;
        const ct=tasks.filter(t=>t.category===cat.id).length;
        const p=ct?Math.round(cd/ct*100):0;
        return `<div class="cat-bar-wrap">
          <div class="cat-bar-label"><span>${cat.icon} ${cat.label}</span><span>${cd}/${ct}</span></div>
          <div class="cat-bar"><div class="cat-bar-fill" style="width:${p}%;background:${cat.color}"></div></div>
        </div>`;}).join("")}
    </div>`;
  }

  if(tab==="diary"){
    html+=`<div class="card">
      <div class="card-title">Заметка — ${today.toLocaleDateString("ru-RU",{day:"numeric",month:"long"})}</div>
      <textarea id="diary-input" placeholder="Что получилось? Что нет? Мысль о себе...">${diary}</textarea>
      <div class="save-hint">Сохраняется автоматически</div>
    </div>
    <div class="card"><div class="card-title">Итог дня</div>
      ${[["Выполнено задач",`${totalDone} из ${totalTasks}`],["XP сегодня",`+${totalDone*10}`],["Всего XP",xp],["Серия",`${streak} дн.`]].map(([l,v])=>`
        <div class="row"><span class="row-label">${l}</span><span class="row-val">${v}</span></div>`).join("")}
    </div>`;
  }

  if(tab==="insights"){
    html+=`<div style="font-size:12px;color:#888;margin-bottom:10px">Разблокировано ${unlockedInsights.length} из ${INSIGHTS.length}</div>`;
    html+=INSIGHTS.map((ins,i)=>{
      const cat=catInfo(ins.cat);
      const unlocked=unlockedInsights.includes(i);
      return `<div class="insight-item ${unlocked?"":"locked"}">
        <div class="badges" style="margin-bottom:${unlocked?6:0}px">
          <span class="badge" style="background:${cat?.color};color:${cat?.text}">${cat?.icon} ${cat?.label}</span>
          ${!unlocked?'<span style="font-size:11px;color:#bbb">🔒 ещё не открыт</span>':''}
        </div>
        ${unlocked?`<div style="font-size:13px;line-height:1.6">${ins.text}</div>`:'<div style="font-size:12px;color:#bbb;margin-top:4px">Продолжай прокачиваться</div>'}
      </div>`;
    }).join("");
  }

  document.getElementById("app").innerHTML=html;

  // diary listener
  const di=document.getElementById("diary-input");
  if(di){
    di.addEventListener("input",e=>saveDiary(e.target.value));
    // prevent re-render killing focus
    di.addEventListener("keydown",e=>e.stopPropagation());
  }
}

render();
</script>
